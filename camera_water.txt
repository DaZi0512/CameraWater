  function compile(type, src, label) {
    const sh = gl.createShader(type);
    gl.shaderSource(sh, src);
    gl.compileShader(sh);

    if (!gl.getShaderParameter(sh, gl.COMPILE_STATUS)) {
      const info = gl.getShaderInfoLog(sh) || "无详细日志";
      const kind = type === gl.VERTEX_SHADER ? "Vertex" : "Fragment";
      showError(kind + " shader编译失败：" + label + "\n\n" + info + "\n\n" + numberSource(src));
      gl.deleteShader(sh);
      return null;
    }
    return sh;
  }

  function makeProgram(vsSrc, fsSrc, label) {
    const vs = compile(gl.VERTEX_SHADER, vsSrc, label);
    const fs = compile(gl.FRAGMENT_SHADER, fsSrc, label);
    if (!vs || !fs) return null;

    const p = gl.createProgram();
    gl.attachShader(p, vs);
    gl.attachShader(p, fs);
    gl.linkProgram(p);

    gl.deleteShader(vs);
    gl.deleteShader(fs);

    if (!gl.getProgramParameter(p, gl.LINK_STATUS)) {
      showError("Shader链接失败：" + label + "\n\n" + (gl.getProgramInfoLog(p) || "无详细日志"));
      gl.deleteProgram(p);
      return null;
    }
    return p;
  }

  function waitForVideoReady() {

      return Promise.resolve();
    }
    return new Promise((resolve, reject) => {
      const start = performance.now();

          resolve();
          return;

        if (performance.now() - start > 5000) {
          reject(new Error("摄像头视频帧超时，未拿到可用videoWidth/videoHeight"));
          return;
        }
        requestAnimationFrame(tick);
      };
      tick();
    });

  async function initHands() {

      showError("MediaPipe Hands加载失败。请检查网络是否能访问：\nhttps://cdn.jsdelivr.net/npm/@mediapipe/hands@0.4/");
      return false;

      hands = new Hands({
        locateFile: file => "https://cdn.jsdelivr.net/npm/@mediapipe/hands@0.4/" + file
      });

        maxNumHands: 2,
        modelComplexity: 1,
        minDetectionConfidence: 0.62,
        minTrackingConfidence: 0.55

        await hands.initialize();
      }

      handsReady = true;
      handsInitialized = true;
      handsFailCount = 0;
      clearError();
      return true;
    } catch (err) {

      handsInitialized = false;

  function videoToScreenUV(xVideoDisplayed, yVideoTop) {
    const vw = video.videoWidth || 1280;
    const vh = video.videoHeight || 720;
    const va = vw / vh;
    const ca = canvas.width / canvas.height;

    let sx = xVideoDisplayed;
    let syTop = yVideoTop;

    if (ca > va) {
      syTop = (yVideoTop - 0.5) * (ca / va) + 0.5;
    } else {
      sx = (xVideoDisplayed - 0.5) * (va / ca) + 0.5;
    }

    return [sx, 1.0 - syTop];
  }

  function lmDist2(a, b) {
    const dx = a.x - b.x;
    const dy = a.y - b.y;
    return Math.sqrt(dx * dx + dy * dy);
  }

  function smooth01(edge0, edge1, x) {
    const t = clamp01((x - edge0) / Math.max(edge1 - edge0, 1e-6));
    return t * t * (3.0 - 2.0 * t);
  }

  function palmInfo(hand) {
    const ids = [0, 5, 9, 13, 17];
    let x = 0.0;
    let y = 0.0;
    for (let i = 0; i < ids.length; i++) {
      x += hand[ids[i]].x;
      y += hand[ids[i]].y;
    }
    const center = {x: x / ids.length, y: y / ids.length};
    const wrist = hand[0];
    const scale = Math.max(0.035,
      0.55 * lmDist2(wrist, hand[9]) +
      0.30 * lmDist2(hand[5], hand[17]) +
      0.15 * lmDist2(hand[5], hand[9])
    );
    return {center, scale};
  }

  function fingerOpenScore(hand, joint, palm) {
    const tip = hand[joint.tip];
    const dip = hand[joint.dip];
    const pip = hand[joint.pip];
    const mcp = hand[joint.mcp];

    const tipPalm = lmDist2(tip, palm.center);
    const pipPalm = lmDist2(pip, palm.center);
    const tipMcp = lmDist2(tip, mcp);
    const pipMcp = lmDist2(pip, mcp);

    const ux = tip.x - dip.x;
    const uy = tip.y - dip.y;
    const vx = dip.x - pip.x;
    const vy = dip.y - pip.y;
    const alignment = (ux * vx + uy * vy) / (Math.sqrt(ux * ux + uy * uy) * Math.sqrt(vx * vx + vy * vy) + 1e-6);

    // 非拇指：指尖要明显远离掌心，并且tip-mcp长度要大于pip-mcp。
    // 握拳时MediaPipe仍会给出tip坐标，但这些坐标靠近掌心，会被这里过滤。
    const normalLength = smooth01(0.58, 0.94, tipMcp / palm.scale);
    const normalOutward = smooth01(1.04, 1.24, tipPalm / Math.max(pipPalm, 1e-6));
    const normalStraight = smooth01(-0.15, 0.38, alignment);

    // 拇指横向活动大，不能完全照搬其它四指，所以阈值略放宽，但仍要求远离掌心。
    const thumbLength = smooth01(0.50, 0.86, tipMcp / palm.scale);
    const thumbOutward = smooth01(1.02, 1.18, tipPalm / Math.max(pipPalm, 1e-6));

    const normalScore = normalLength * normalOutward * (0.72 + 0.28 * normalStraight);
    const thumbScore = thumbLength * thumbOutward;

    return joint.thumb > 0.5 ? thumbScore : normalScore;
  }

  function onHandsResults(results) {
    const now = performance.now() * 0.001;
    let index = 0;

    const allHands = results && results.multiHandLandmarks ? results.multiHandLandmarks : [];
    for (let h = 0; h < allHands.length && index < MAX_TIPS; h++) {
      const hand = allHands[h];
      const palm = palmInfo(hand);

      for (let k = 0; k < FINGER_JOINTS.length && index < MAX_TIPS; k++) {
        const joint = FINGER_JOINTS[k];
        const lm = hand[joint.tip];
        const xDisplayed = isFrontCamera ? 1.0 - lm.x : lm.x;
        const uv = videoToScreenUV(xDisplayed, lm.y);

        const insideX = step01(uv[0]) * step01(1.0 - uv[0]);
        const insideY = step01(uv[1]) * step01(1.0 - uv[1]);
        const inside = insideX * insideY;

        const t = tips[index];
        const oldX = t.x;
        const oldY = t.y;
        const rawDx = uv[0] - oldX;
        const rawDy = uv[1] - oldY;
        const rawSpeed = Math.sqrt(rawDx * rawDx + rawDy * rawDy);

        const openScore = fingerOpenScore(hand, joint, palm);
        const openGate = smooth01(0.42, 0.70, openScore);

        // 关键修改：必须是“伸开的指尖 + 发生移动”才注入波纹。
        // 这样握拳、手掌、关节、误识别点不会产生大面积波纹。
        const motionGate = smooth01(0.0018, 0.016, rawSpeed);
        const close = clamp01((-lm.z - 0.010) / 0.105);
        const gate = inside * openGate * motionGate;

        if (gate < 0.025 || t.active < 0.04) {
          t.px = uv[0];
          t.py = uv[1];
        } else {
          t.px = oldX;
          t.py = oldY;
        }

        t.x = uv[0];
        t.y = uv[1];

        const dx = t.x - t.px;
        const dy = t.y - t.py;
        const speed = Math.sqrt(dx * dx + dy * dy);

        t.active = gate;
        t.strength = gate * clamp01(0.14 + close * 0.20 + speed * 24.0);
        t.lastSeen = now;

        index++;
      }
    }

    for (; index < MAX_TIPS; index++) {
      tips[index].active *= 0.72;
      tips[index].strength *= 0.68;
    }
  }

  function step01(v) {
    return v >= 0 ? 1 : 0;
  }

  function clamp01(v) {
    return Math.max(0, Math.min(1, v));
  }

  async function detectLoop(t) {
    requestAnimationFrame(detectLoop);

    if (!handsReady || !hands || handsResetting || !video.videoWidth || !video.videoHeight || video.readyState < 2) return;

    const minInterval = 1000 / 24;
    if (t - lastDetectTime < minInterval) return;

    // 关键修复2：MediaPipe Solution不是可重入的，上一帧没处理完时不能继续send。
    // 旧版本会因为并发send触发内部buffer为undefined。
    if (handsSending) return;

    lastDetectTime = t;
    handsSending = true;

    try {
      await hands.send({image: video});
      handsFailCount = 0;
      if (errorBox.textContent.indexOf("MediaPipe Hands运行失败") === 0) clearError();
    } catch (err) {
      handsFailCount += 1;
      const msg = err && err.message ? err.message : String(err);
      console.warn("MediaPipe Hands运行失败：", msg);

      // 偶发首帧错误不直接打断画面；连续失败才提示。
      if (handsFailCount >= 3) {
        showError(
          "MediaPipe Hands运行失败：\n" + msg +
          "\n\n已自动尝试重置手势识别。若仍失败，多半是CDN模型/wasm没有完整加载，刷新页面或换Chrome/Safari再试。"
        );
        resetHands();
      }
    } finally {
      handsSending = false;
    }
  }

  function updateVideoTexture() {
    if (!video.videoWidth || video.readyState < 2) return;
    gl.activeTexture(gl.TEXTURE1);
    gl.bindTexture(gl.TEXTURE_2D, videoTexture);
    gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true);
    try {
      gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, video);
    } catch (err) {
      // 有些浏览器在视频第一帧尚未准备好时会短暂报错，忽略即可。
    }
    gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false);
  }

  function fillTipUniforms() {
    for (let i = 0; i < MAX_TIPS; i++) {
      const t = tips[i];
      tipUniform[i * 2] = t.x;
      tipUniform[i * 2 + 1] = t.y;
      prevUniform[i * 2] = t.px;
      prevUniform[i * 2 + 1] = t.py;
      activeUniform[i] = t.active;
      strengthUniform[i] = t.strength;
    }
  }

  function simStep(timeSec) {
    fillTipUniforms();

    gl.useProgram(simProgram);
    bindQuad(simProgram);

    gl.bindFramebuffer(gl.FRAMEBUFFER, writeState.fbo);
    gl.viewport(0, 0, simW, simH);

    gl.activeTexture(gl.TEXTURE0);
    gl.bindTexture(gl.TEXTURE_2D, readState.tex);
    gl.uniform1i(simU.state, 0);
    gl.uniform2f(simU.texel, 1 / simW, 1 / simH);
    gl.uniform1f(simU.aspect, simW / simH);
    gl.uniform1f(simU.time, timeSec);
    gl.uniform1f(simU.mode, mode);
    gl.uniform2fv(simU.tip, tipUniform);
    gl.uniform2fv(simU.prev, prevUniform);
    gl.uniform1fv(simU.active, activeUniform);
    gl.uniform1fv(simU.strength, strengthUniform);

    gl.drawArrays(gl.TRIANGLES, 0, 6);

    const tmp = readState;
    readState = writeState;
    writeState = tmp;
  }

  function render(timeSec) {
    resize();
    updateVideoTexture();

    // 两次子步进，涟漪扩散更顺，手机上仍然是WebGL1负载。
    simStep(timeSec);
    simStep(timeSec + 0.004);

    gl.useProgram(renderProgram);
    bindQuad(renderProgram);

    gl.bindFramebuffer(gl.FRAMEBUFFER, null);
    gl.viewport(0, 0, canvas.width, canvas.height);

    gl.activeTexture(gl.TEXTURE0);
    gl.bindTexture(gl.TEXTURE_2D, readState.tex);
    gl.uniform1i(renderU.state, 0);

    gl.activeTexture(gl.TEXTURE1);
    gl.bindTexture(gl.TEXTURE_2D, videoTexture);
    gl.uniform1i(renderU.video, 1);

    gl.uniform2f(renderU.texel, 1 / simW, 1 / simH);
    gl.uniform2f(renderU.resolution, canvas.width, canvas.height);
    gl.uniform2f(renderU.videoResolution, video.videoWidth || 1280, video.videoHeight || 720);
    gl.uniform1f(renderU.mirror, isFrontCamera ? 1.0 : 0.0);
    gl.uniform1f(renderU.mode, mode);
    gl.uniform1f(renderU.time, timeSec);

    gl.drawArrays(gl.TRIANGLES, 0, 6);

    requestAnimationFrame(t => render(t * 0.001));
  }

  const SIM_FRAG = `
precision highp float;

varying vec2 vUv;

uniform sampler2D uState;
uniform vec2 uTexel;
uniform float uAspect;
uniform float uTime;
uniform float uMode;

uniform vec2 uTip[10];
uniform vec2 uPrevTip[10];
uniform float uTipActive[10];
uniform float uTipStrength[10];

float decodeH(vec4 c){
  return c.r * 2.0 - 1.0;
}

float decodePrev(vec4 c){
  return c.g * 2.0 - 1.0;
}

float capsuleSDF(vec2 p, vec2 a, vec2 b, float r){
  vec2 pa = p - a;
  vec2 ba = b - a;
  float h = clamp(dot(pa, ba) / (dot(ba, ba) + 1e-5), 0.0, 1.0);
  return length(pa - ba * h) - r;
}

void main(){
  vec2 e = uTexel;

  vec4 c0 = texture2D(uState, vUv);
  float h  = decodeH(c0);
  float hp = decodePrev(c0);

  float hL  = decodeH(texture2D(uState, vUv + vec2(-e.x, 0.0)));
  float hR  = decodeH(texture2D(uState, vUv + vec2( e.x, 0.0)));
  float hB  = decodeH(texture2D(uState, vUv + vec2(0.0, -e.y)));
  float hT  = decodeH(texture2D(uState, vUv + vec2(0.0,  e.y)));

  float hLB = decodeH(texture2D(uState, vUv + vec2(-e.x, -e.y)));
  float hRB = decodeH(texture2D(uState, vUv + vec2( e.x, -e.y)));
  float hLT = decodeH(texture2D(uState, vUv + vec2(-e.x,  e.y)));
  float hRT = decodeH(texture2D(uState, vUv + vec2( e.x,  e.y)));

  // 9-tap isotropic Laplacian: 比4邻域更不容易出现方形波纹。
  float lap = (4.0 * (hL + hR + hB + hT) + (hLB + hRB + hLT + hRT) - 20.0 * h) * 0.1666667;

  float speed = mix(0.50, 0.62, uMode);
  float nextH = 2.0 * h - hp + lap * speed;

  // 非线性衰减：小振幅衰减快，大振幅衰减慢。
  float amp = abs(h);
  float inertia = smoothstep(0.018, 0.18, amp);
  float dampingSmall = mix(0.942, 0.934, uMode);
  float dampingLarge = mix(0.988, 0.982, uMode);
  float damping = mix(dampingSmall, dampingLarge, inertia);
  nextH *= damping;

  vec2 p = vec2(vUv.x * uAspect, vUv.y);
  float impulse = 0.0;

  for(int i = 0; i < 10; i++){
    vec2 a = vec2(uPrevTip[i].x * uAspect, uPrevTip[i].y);
    vec2 b = vec2(uTip[i].x * uAspect, uTip[i].y);
    float s = clamp(uTipStrength[i], 0.0, 1.0);
    float r = mix(0.008, 0.023, s) + uMode * 0.003;

    float d = capsuleSDF(p, a, b, r);
    float core = exp(-(d * d) / (r * r + 1e-5));
    float wide = exp(-(d * d) / (r * r * 5.8 + 1e-5));
    float ringBias = core - 0.26 * wide;

    // 低频扰动让静止指尖也有轻微“贴在水膜上”的感觉。
    float pulse = 0.78 + 0.22 * sin(uTime * 9.0 + float(i) * 1.73);
    impulse += uTipActive[i] * s * ringBias * pulse;
  }

  nextH += impulse * mix(0.035, 0.048, uMode);

  // 边界吸收，减少墙面反弹。
  vec2 edge = smoothstep(vec2(0.0), vec2(0.055), vUv) * smoothstep(vec2(0.0), vec2(0.055), 1.0 - vUv);
  float edgeFade = edge.x * edge.y;
  nextH *= mix(0.86, 1.0, edgeFade);

  nextH = clamp(nextH, -0.82, 0.82);

  gl_FragColor = vec4(nextH * 0.5 + 0.5, h * 0.5 + 0.5, lap * 0.5 + 0.5, 1.0);
}
`;

  const RENDER_FRAG = `
precision highp float;

varying vec2 vUv;

uniform sampler2D uState;
uniform sampler2D uVideo;
uniform vec2 uTexel;
uniform vec2 uResolution;
uniform vec2 uVideoResolution;
uniform float uMirror;
uniform float uMode;
uniform float uTime;

float decodeH(vec4 c){
  return c.r * 2.0 - 1.0;
}

vec2 coverUV(vec2 uv){
  float ca = uResolution.x / max(uResolution.y, 1.0);
  float va = uVideoResolution.x / max(uVideoResolution.y, 1.0);

  vec2 videoWide = vec2((uv.x - 0.5) * (ca / va) + 0.5, uv.y);
  vec2 canvasWide = vec2(uv.x, (uv.y - 0.5) * (va / ca) + 0.5);
  float canvasIsWide = step(va, ca);

  vec2 outUV = mix(videoWide, canvasWide, canvasIsWide);
  outUV.x = mix(outUV.x, 1.0 - outUV.x, uMirror);
  return clamp(outUV, 0.001, 0.999);
}

vec3 videoSample(vec2 uv){
  return texture2D(uVideo, coverUV(uv)).rgb;
}

void main(){
  vec2 e = uTexel;

  float h  = decodeH(texture2D(uState, vUv));
  float hL = decodeH(texture2D(uState, vUv + vec2(-e.x, 0.0)));
  float hR = decodeH(texture2D(uState, vUv + vec2( e.x, 0.0)));
  float hB = decodeH(texture2D(uState, vUv + vec2(0.0, -e.y)));
  float hT = decodeH(texture2D(uState, vUv + vec2(0.0,  e.y)));

  float hLB = decodeH(texture2D(uState, vUv + vec2(-e.x, -e.y)));
  float hRB = decodeH(texture2D(uState, vUv + vec2( e.x, -e.y)));
  float hLT = decodeH(texture2D(uState, vUv + vec2(-e.x,  e.y)));
  float hRT = decodeH(texture2D(uState, vUv + vec2( e.x,  e.y)));

  vec2 grad = vec2(hR - hL, hT - hB);
  float lap = (4.0 * (hL + hR + hB + hT) + (hLB + hRB + hLT + hRT) - 20.0 * h) * 0.1666667;

  // 折射：直接使用坡度grad，不归一化；坡越陡，偏移越大。
  float refractScale = mix(0.072, 0.092, uMode);
  float lensScale = mix(0.082, 0.116, uMode);

  // Laplacian曲率项：波峰/波谷有局部放大镜式变形。
  vec2 offset = grad * refractScale + grad * lap * lensScale;

  float chroma = mix(0.42, 0.72, uMode);
  vec3 col;
  col.r = videoSample(vUv + offset * (1.0 + 0.90 * chroma)).r;
  col.g = videoSample(vUv + offset).g;
  col.b = videoSample(vUv + offset * (1.0 - 0.55 * chroma)).b;

  vec3 normal = normalize(vec3(-grad.x * 34.0, -grad.y * 34.0, 1.0));
  vec3 viewDir = vec3(0.0, 0.0, 1.0);
  vec3 lightDir = normalize(vec3(-0.42, 0.52, 0.74));
  vec3 halfDir = normalize(lightDir + viewDir);

  // Schlick Fresnel：掠射角反光更强。
  float ndv = clamp(dot(normal, viewDir), 0.0, 1.0);
  float fresnel = 0.035 + (1.0 - 0.035) * pow(1.0 - ndv, 5.0);

  // 双瓣高光：尖锐specular + 宽sheen。
  float ndh = max(dot(normal, halfDir), 0.0);
  float sharpSpec = pow(ndh, mix(90.0, 150.0, uMode));
  float broadSheen = pow(max(dot(normal, lightDir), 0.0), mix(8.0, 14.0, uMode));

  float slope = length(grad);
  float rim = smoothstep(0.028, 0.135, slope) + smoothstep(0.044, 0.160, abs(lap));

  vec3 warm = vec3(1.0, 0.38, 0.18);
  vec3 cold = vec3(0.12, 0.34, 0.92);

  // 波深明暗：波峰暖亮，波谷冷暗；幅度控制得比较克制，避免整片蓝/黄污染画面。
  col += warm * max(h, 0.0) * mix(0.22, 0.32, uMode);
  col -= cold * max(-h, 0.0) * mix(0.10, 0.15, uMode);

  vec3 reflectTint = mix(vec3(0.82, 0.95, 1.0), vec3(1.0, 0.82, 0.72), smoothstep(0.0, 0.16, h));
  col += reflectTint * fresnel * mix(0.42, 0.58, uMode);
  col += vec3(1.0, 0.92, 0.78) * sharpSpec * mix(1.18, 1.62, uMode);
  col += vec3(0.80, 0.92, 1.0) * broadSheen * mix(0.14, 0.22, uMode);

  // 类似参考图的银白水膜边缘，但不做半透明覆盖。
  vec3 edgeWhite = mix(vec3(0.92, 1.0, 1.0), vec3(1.0, 0.86, 0.78), uMode);
  col += edgeWhite * rim * mix(0.13, 0.23, uMode);

  // 轻微暗角，把注意力压到手和水纹区域。
  vec2 q = vUv - 0.5;
  float vignette = smoothstep(0.88, 0.20, dot(q, q));
  col *= mix(0.82, 1.05, vignette);

  // 细小晶体折线感，只在CRYSTAL模式增强。
  float crystalLine = sin((vUv.x + vUv.y) * 720.0 + h * 80.0 + uTime * 0.65);
  col += vec3(1.0) * smoothstep(0.985, 1.0, crystalLine) * uMode * rim * 0.16;

  col = pow(max(col, 0.0), vec3(0.96));
  gl_FragColor = vec4(clamp(col, 0.0, 1.0), 1.0);
}
`;

  if (initGL()) {
    startCamera();
    requestAnimationFrame(t => render(t * 0.001));
  }
})();
